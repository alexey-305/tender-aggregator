import { useState } from "react";
import { X } from "lucide-react";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { Switch } from "@/components/ui/switch";
import { Checkbox } from "@/components/ui/checkbox";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogFooter,
} from "@/components/ui/dialog";
import { Badge } from "@/components/ui/badge";

interface FilterTemplateModalProps {
  open: boolean;
  onOpenChange: (open: boolean) => void;
}

export default function FilterTemplateModal({ open, onOpenChange }: FilterTemplateModalProps) {
  const [templateName, setTemplateName] = useState("");
  const [notify, setNotify] = useState(true);
  const [includeWords, setIncludeWords] = useState<string[]>([]);
  const [excludeWords, setExcludeWords] = useState<string[]>([]);
  const [newWord, setNewWord] = useState("");

  const addWord = (type: "include" | "exclude") => {
    if (!newWord.trim()) return;
    if (type === "include") {
      setIncludeWords([...includeWords, newWord.trim()]);
    } else {
      setExcludeWords([...excludeWords, newWord.trim()]);
    }
    setNewWord("");
  };

  const removeWord = (word: string, type: "include" | "exclude") => {
    if (type === "include") {
      setIncludeWords(includeWords.filter(w => w !== word));
    } else {
      setExcludeWords(excludeWords.filter(w => w !== word));
    }
  };

  const handleSave = () => {
    console.log("Шаблон сохранён:", { templateName, includeWords, excludeWords, notify });
    onOpenChange(false);
  };

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="max-w-4xl max-h-[90vh] overflow-y-auto">
        <DialogHeader>
          <DialogTitle className="text-2xl">Новый шаблон фильтра</DialogTitle>
        </DialogHeader>

        <div className="grid grid-cols-2 gap-8 py-4">
          <div className="space-y-6">
            <div>
              <Label>Название шаблона</Label>
              <Input 
                value={templateName} 
                onChange={(e) => setTemplateName(e.target.value)}
                placeholder="Закупки металлоконструкций в ЦФО" 
              />
            </div>

            <div className="flex items-center justify-between">
              <Label>Уведомления о новых закупках</Label>
              <Switch checked={notify} onCheckedChange={setNotify} />
            </div>

            <div>
              <Label>Включающие слова</Label>
              <div className="flex gap-2">
                <Input 
                  value={newWord} 
                  onChange={(e) => setNewWord(e.target.value)}
                  onKeyDown={(e) => e.key === "Enter" && addWord("include")}
                  placeholder="металлоконструкции" 
                />
                <Button onClick={() => addWord("include")}>Добавить</Button>
              </div>
              <div className="flex flex-wrap gap-2 mt-3">
                {includeWords.map(word => (
                  <Badge key={word} variant="default" className="cursor-pointer" onClick={() => removeWord(word, "include")}>
                    {word} <X className="ml-1 h-3 w-3" />
                  </Badge>
                ))}
              </div>
            </div>
          </div>

          <div className="space-y-6">
            <div>
              <Label>Исключающие слова</Label>
              <div className="flex gap-2">
                <Input 
                  value={newWord} 
                  onChange={(e) => setNewWord(e.target.value)}
                  onKeyDown={(e) => e.key === "Enter" && addWord("exclude")}
                  placeholder="бумага" 
                />
                <Button onClick={() => addWord("exclude")}>Добавить</Button>
              </div>
              <div className="flex flex-wrap gap-2 mt-3">
                {excludeWords.map(word => (
                  <Badge key={word} variant="secondary" className="cursor-pointer" onClick={() => removeWord(word, "exclude")}>
                    {word} <X className="ml-1 h-3 w-3" />
                  </Badge>
                ))}
              </div>
            </div>
          </div>
        </div>

        <DialogFooter>
          <Button variant="outline" onClick={() => onOpenChange(false)}>Отменить</Button>
          <Button onClick={handleSave} size="lg">Сохранить и применить</Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
}
